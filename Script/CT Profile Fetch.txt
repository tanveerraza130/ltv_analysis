function getEventsFromCleverTap() {
  // Get the sheet and verify it exists
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Get_Events");
  if (!sheet) {
    SpreadsheetApp.getUi().alert("Error: 'Get_Events' sheet not found!");
    return;
  }

  // Get CleverTap configuration
  try {
    var config = getCleverTapConfig();
  } catch (e) {
    SpreadsheetApp.getUi().alert("Error with CleverTap configuration: " + e.message);
    return;
  }

  const ui = SpreadsheetApp.getUi();
  
  // Get input parameters
  const dataRange = sheet.getDataRange().getValues();
  if (dataRange.length <= 1) {
    ui.alert("Please fill in the 'Get_Events' sheet with Event Name, From Date, and To Date.");
    return;
  }
  
  // Extract event parameters (assuming data is in first row after header)
  const eventParams = dataRange[1]; // Row index 1 (second row)
  if (!eventParams[0] || !eventParams[1] || !eventParams[2]) {
    ui.alert("Please provide Event Name, From Date (YYYYMMDD), and To Date (YYYYMMDD).");
    return;
  }
  
  const eventName = eventParams[0].toString().trim();
  const fromDate = parseInt(eventParams[1].toString().replace(/[^\d]/g, ""), 10);
  const toDate = parseInt(eventParams[2].toString().replace(/[^\d]/g, ""), 10);
  
  // Validate date formats
  if (isNaN(fromDate) || isNaN(toDate) || 
      fromDate.toString().length !== 8 || toDate.toString().length !== 8) {
    ui.alert("Date format should be YYYYMMDD (e.g., 20230101)");
    return;
  }
  
  // Optional parameters
  const batchSize = eventParams[3] ? parseInt(eventParams[3]) : 50;
  
  // Show loading indicator
  sheet.getRange(2, 5).setValue("Loading data from CleverTap...");
  
  // Step 1: Get cursor
  Logger.log(`Fetching events for "${eventName}" from ${fromDate} to ${toDate}`);
  
  const postUrl = `https://${config.Region}.api.clevertap.com/1/events.json?batch_size=${batchSize}&app=true`;
  const postPayload = {
    event_name: eventName,
    from: fromDate,
    to: toDate
  };
  
  const options = {
    method: "post",
    contentType: "application/json",
    headers: config.headers,
    payload: JSON.stringify(postPayload),
    muteHttpExceptions: true
  };
  
  try {
    // Make the initial request to get the cursor
    const postResponse = UrlFetchApp.fetch(postUrl, options);
    const responseCode = postResponse.getResponseCode();
    
    if (responseCode !== 200) {
      ui.alert(`API error: HTTP ${responseCode}\n${postResponse.getContentText()}`);
      sheet.getRange(2, 5).clearContent();
      return;
    }
    
    const postData = JSON.parse(postResponse.getContentText());
    if (postData.status !== "success" || !postData.cursor) {
      ui.alert(`Failed to get cursor: ${postData.error || "Unknown error"}`);
      sheet.getRange(2, 5).clearContent();
      return;
    }
    
    // Step 2: Fetch events using cursor
    let cursor = postData.cursor;
    let allRecords = [];
    let batch = 1;
    let totalRecords = 0;
    const maxRetries = 3;
    
    sheet.getRange(2, 5).setValue("Retrieving events (batch 1)...");
    
    while (cursor) {
      Logger.log(`Fetching batch ${batch} with cursor`);
      
      // IMPORTANT FIX: Use POST method for cursor to avoid URL length issues
      const getUrl = `https://${config.Region}.api.clevertap.com/1/events.json`;
      const getOptions = {
        method: "post", // Changed from GET to POST
        contentType: "application/json",
        headers: config.headers,
        payload: JSON.stringify({ cursor: cursor }), // Send cursor in the payload instead of URL
        muteHttpExceptions: true
      };
      
      let retries = 0;
      let getData = null;
      
      // Implement retry logic
      while (retries < maxRetries) {
        try {
          const getResponse = UrlFetchApp.fetch(getUrl, getOptions);
          const responseCode = getResponse.getResponseCode();
          
          if (responseCode !== 200) {
            Logger.log(`Batch ${batch} HTTP error: ${responseCode}`);
            retries++;
            Utilities.sleep(2000); // Wait 2 seconds before retrying
            continue;
          }
          
          getData = JSON.parse(getResponse.getContentText());
          
          // If data is not ready yet, wait and retry
          if (getData.status === "fail" && getData.code === 2) {
            Logger.log(`Batch ${batch} not ready yet, retrying...`);
            retries++;
            sheet.getRange(2, 5).setValue(`Waiting for batch ${batch} data...`);
            Utilities.sleep(3000); // Wait 3 seconds before retrying
            continue;
          }
          
          // Break out of retry loop if successful
          break;
          
        } catch (e) {
          Logger.log(`Error in batch ${batch}: ${e.toString()}`);
          retries++;
          Utilities.sleep(2000);
        }
      }
      
      // If all retries failed
      if (retries >= maxRetries || !getData || getData.status !== "success") {
        Logger.log(`Failed to retrieve batch ${batch} after ${maxRetries} attempts`);
        if (getData && getData.error) {
          Logger.log(`Error: ${getData.error}`);
        }
        break;
      }
      
      // Process records in this batch
      if (getData.records && getData.records.length > 0) {
        allRecords = allRecords.concat(getData.records);
        totalRecords += getData.records.length;
        sheet.getRange(2, 5).setValue(`Retrieved ${totalRecords} events (batch ${batch})...`);
      }
      
      // Update cursor for next batch
      cursor = getData.next_cursor || null;
      batch++;
      
      // Update progress indicator
      if (cursor) {
        sheet.getRange(2, 5).setValue(`Retrieving events (batch ${batch})...`);
      }
    }
    
    // Clear old output
    const outputStartRow = 5;
    const outputStartCol = 1;
    if (sheet.getLastRow() > outputStartRow) {
      sheet.getRange(outputStartRow, outputStartCol, 
                     sheet.getLastRow() - outputStartRow + 1, 
                     sheet.getMaxColumns()).clearContent();
    }
    
    // Write results header
    const outputHeader = [
      "Object ID", 
      "Identity", 
      "Email",
      "Platform",
      "Event Name", 
      "Event Time (UTC)", 
      "Formatted Time", 
      "Event Properties", 
      "Session Properties",
      "Device/App Details"
    ];
    
    sheet.getRange(outputStartRow, outputStartCol, 1, outputHeader.length)
         .setValues([outputHeader])
         .setFontWeight("bold")
         .setBackground("#f3f3f3");
    
    // Format and write results
    if (allRecords.length > 0) {
      const outputData = [];
      
      allRecords.forEach(rec => {
        const profile = rec.profile || {};
        const eventProps = rec.event_props || {};
        const sessionProps = rec.session_props || {};
        
        // Convert timestamp to readable format if possible
        let tsString = rec.ts ? rec.ts.toString() : "";
        let formattedTime = "";
        
        if (tsString.length === 14) { // Format: YYYYMMDDHHmmSS
          try {
            const year = tsString.substring(0, 4);
            const month = tsString.substring(4, 6);
            const day = tsString.substring(6, 8);
            const hour = tsString.substring(8, 10);
            const minute = tsString.substring(10, 12);
            const second = tsString.substring(12, 14);
            
            const date = new Date(Date.UTC(year, month-1, day, hour, minute, second));
            formattedTime = date.toLocaleString();
          } catch (e) {
            formattedTime = "";
          }
        }
        
        // Prepare device/app details
        const deviceDetails = {};
        if (profile.os_version) deviceDetails.os_version = profile.os_version;
        if (profile.device_model) deviceDetails.device_model = profile.device_model;
        if (profile.app_version) deviceDetails.app_version = profile.app_version;
        if (profile.device_make) deviceDetails.device_make = profile.device_make;
        
        // Create a row for this event record
        outputData.push([
          profile.objectId || "",
          profile.identity || "",
          profile.email || "",
          profile.platform || "",
          eventName,
          tsString,
          formattedTime,
          JSON.stringify(eventProps, null, 2),
          JSON.stringify(sessionProps, null, 2),
          Object.keys(deviceDetails).length > 0 ? JSON.stringify(deviceDetails, null, 2) : ""
        ]);
      });
      
      // Write data to sheet
      if (outputData.length > 0) {
        sheet.getRange(outputStartRow + 1, outputStartCol, outputData.length, outputHeader.length)
             .setValues(outputData);
        
        // Auto-resize columns for better readability
        for (let i = 1; i <= outputHeader.length; i++) {
          sheet.autoResizeColumn(i);
        }
      }
      
      // Final status indicator
      sheet.getRange(2, 5).setValue(`✅ Complete: Fetched ${allRecords.length} events across ${batch-1} batch(es).`);
      ui.alert(`✅ Events fetched: ${allRecords.length} records across ${batch-1} batch(es).`);
    } else {
      sheet.getRange(outputStartRow + 1, outputStartCol)
           .setValue("No events found for the specified parameters.");
      sheet.getRange(2, 5).setValue("✅ Complete: No events found.");
      ui.alert("No events returned for the specified parameters.");
    }
    
  } catch (error) {
    Logger.log(`Error: ${error.toString()}`);
    sheet.getRange(2, 5).clearContent();
    ui.alert(`Error fetching events: ${error.toString()}`);
  }
}

function getCleverTapConfig() {
  const ui = SpreadsheetApp.getUi();
  var master = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Master");
  if (!master) {
    throw new Error("Master sheet not found!");
  }
  const accountId = "6K5-56W-8K7Z";
  const passcode = "WHO-JIE-CLEL";
  let region = "in1"
  // Set default region to 'eu1' if not provided
  if (!region) {
    region = "in1";
    const response = ui.alert("Region not set. Set the Region to 'eu1'. Continue?", ui.ButtonSet.YES_NO);
    if (response === ui.Button.NO) {
      Logger.log(":octagonal_sign: User aborted due to Region Not Selected.");
      throw new Error("Operation cancelled by user");
    }
    Logger.log("Region not set. Defaulting to 'eu1'");
  }
  if (!accountId) {
    throw new Error("Missing CleverTap Account ID in Master sheet.");
  }
  if (!passcode) {
    throw new Error("Missing CleverTap Passcode in Master sheet.");
  }
  return {
    headers: {
      "X-CleverTap-Account-Id": accountId,
      "X-CleverTap-Passcode": passcode,
      "Content-Type": "application/json"
    },
    "Region": region,
  };
}

// :pushpin: Process Selections dynamically
function processSelections(selectedOptions) {
  const master = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Master");
  const { row: startRow, col: startCol } = findUserIdentitiesCell();
  const allOptions = ["Identity", "Email", "Phone"]; // In future you can expand this
  // Clear old checkboxes and labels
  master.getRange(startRow + 1, startCol, allOptions.length, 2).clearContent().removeCheckboxes();
  // Write new checkboxes and labels
  for (let i = 0; i < allOptions.length; i++) {
    const checkboxCell = master.getRange(startRow + 1 + i, startCol);
    const labelCell = master.getRange(startRow + 1 + i, startCol + 1);
    checkboxCell.insertCheckboxes();
    const isChecked = selectedOptions.includes(allOptions[i]);
    checkboxCell.setValue(isChecked);
    labelCell.setValue(allOptions[i]);
  }
  return true;
}
function getSelectedIdentity() {
  const master = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Master");
  const { row: userIdentitiesRow, col: userIdentitiesCol } = findUserIdentitiesCell();
  // Define options
  const allOptions = ["Identity", "Email", "Phone"];
  let selectedOptions = [];
  // Start checking from the row below "USER IDENTITIES"
  for (let i = 0; i < allOptions.length; i++) {
    const checkboxCell = master.getRange(userIdentitiesRow + 1 + i, userIdentitiesCol);
    const isChecked = checkboxCell.getValue();
    if (isChecked === true) {
      selectedOptions.push(allOptions[i]);
    }
  }
  Logger.log("Selected options: " + selectedOptions.join(", "));
  return selectedOptions;
}
// onEdit that dynamically re-checks Identity if unchecked
function onEdit(e) {
  const sheet = e.source.getSheetByName("Master");
  if (!sheet) return;
  try {
    const { row: userIdentitiesRow, col: userIdentitiesCol } = findUserIdentitiesCell();
    const editedRow = e.range.getRow();
    const editedCol = e.range.getColumn();
    // Check if user edited the Identity checkbox row
    if (editedRow === userIdentitiesRow + 1 && editedCol === userIdentitiesCol) {
      if (e.value === "FALSE") {
        e.range.setValue(true);
        SpreadsheetApp.getUi().alert(":warning: Identity Mandatory\n\nThe 'Identity' field cannot be unchecked. It's a required parameter.");
      }
    }
  } catch (error) {
    Logger.log("onEdit error: " + error);
  }
}
function findUserIdentitiesCell() {
  const master = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Master");
  const range = master.getDataRange();
  const values = range.getValues();
  for (let row = 0; row < values.length; row++) {
    for (let col = 0; col < values[row].length; col++) {
      if (values[row][col] === 'USER IDENTITIES') {
        return { row: row + 1, col: col + 1 }; // 1-indexed
      }
    }
  }
  throw new Error('USER IDENTITIES not found in Master sheet.');
}
function findCellValue(masterSheet, label) {
  const range = masterSheet.getDataRange();
  const values = range.getValues();
  for (let row = 0; row < values.length; row++) {
    for (let col = 0; col < values[row].length; col++) {
      if (typeof values[row][col] === "string" && values[row][col].trim() === label) {
        // ui.alert(row + 1, col + 2)
        return masterSheet.getRange(row + 1, col + 2).getValue().toString().trim(); // +2 because next column
      }
    }
  }
  throw new Error(`Label "${label}" not found in sheet.`);
}
