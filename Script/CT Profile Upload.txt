function uploadCleverTapProfilesInBatches() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Email");
  const headerRow = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
  const data = sheet.getRange(2, 1, sheet.getLastRow() - 1, headerRow.length).getValues();

  const url = 'https://in1.api.clevertap.com/1/upload';
  const headers = {
    'X-CleverTap-Account-Id': '6K5-56W-8K7Z',
    'X-CleverTap-Passcode': 'WHO-JIE-CLEL',
    'Content-Type': 'application/json; charset=utf-8',
  };

  const batchSize = 500;

  for (let i = 0; i < data.length; i += batchSize) {
    const batch = data.slice(i, i + batchSize);
    const payload = {
      d: []
    };

    batch.forEach(row => {
      const profile = {};
      let identity = "";

      headerRow.forEach((colName, colIndex) => {
        const value = row[colIndex];
        if (colName.toLowerCase() === "identity") {
          identity = String(value).trim();
        } else if (colName && value !== "") {
          profile[colName.trim()] = value;
        }
      });

      if (identity) {
        payload.d.push({
          identity: identity,
          type: "profile",
          profileData: profile
        });
      }
    });

    Logger.log(`📤 Sending batch ${Math.floor(i / batchSize) + 1} with ${payload.d.length} records`);
    Logger.log("🔍 Payload:\n" + JSON.stringify(payload, null, 2));

    try {
      const response = UrlFetchApp.fetch(url, {
        method: "post",
        headers: headers,
        payload: JSON.stringify(payload),
        muteHttpExceptions: true
      });

      const responseText = response.getContentText();
      Logger.log(`📬 Response for batch ${Math.floor(i / batchSize) + 1}:\n` + responseText);
    } catch (e) {
      Logger.log(`❌ Error in batch ${Math.floor(i / batchSize) + 1}: ${e.message}`);
    }

    // To avoid throttling by CleverTap
    Utilities.sleep(500);
  }
}
